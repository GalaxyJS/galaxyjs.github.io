Galaxy.bootstrap

Add this code to your index.html 

(function () {
  window.addEventListener('load', run);

  function run() {
    Galaxy.boot({
      // The path to your main module file
      url: 'modules/main/index.html'
    }, 
    function (module) {
      Galaxy.ui.setContent('body', module.scope.html);
    });
  }
})();
