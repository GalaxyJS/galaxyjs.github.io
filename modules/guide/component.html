Galaxy.bootstrap

Add this code to your index.html 

(function () {
  window.addEventListener('load', run);

  function run() {
    Galaxy.boot({
      url: 'main.html' // The path to your main component file
    }, 
    function (module) {
      Galaxy.ui.setContent('body', module.scope.html);
    });
  }
})();
